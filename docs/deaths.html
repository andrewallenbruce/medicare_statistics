                      names_to = "month",
                      values_to = "deaths_pct") |> 
  dplyr::mutate(month = stringr::str_remove(month, "_pct"),
                month = stringr::str_to_title(month),
                month = factor(month, levels = month.name),
                date = clock::as_date(clock::year_month_day(year, as.integer(month), 1)),
